178  struct handlebars_context * context,
179  struct handlebars_string * string,
180  struct handlebars_token * token,
181  int flags
182 ) {
183  const char * name = handlebars_token_readable_type(token->token);
184  size_t name_len = strlen(name);
185  const char * sep = flags & handlebars_token_print_flag_newlines ? "\n" : " ";
186  struct handlebars_string * text = handlebars_string_addcslashes(context, token->string, HBS_STRL("\r\n\t\v"));
187  string = handlebars_string_append(context, string, name, name_len);
188  string = handlebars_string_append(context, string, HBS_STRL(" ["));
189  string = handlebars_string_append(context, string, text->val, text->len);
190  string = handlebars_string_append(context, string, HBS_STRL("]"));
191  string = handlebars_string_append(context, string, sep, 1);
192  handlebars_talloc_free(text);
193  return string;
194 }
