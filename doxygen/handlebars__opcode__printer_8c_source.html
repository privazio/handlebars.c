75  assert(operand != NULL);
76 
77  switch( operand->type ) {

88  tmp = handlebars_string_addcslashes(context, operand->data.string.string, HBS_STRL("\r\n\t"));
89  string = handlebars_string_asprintf_append(context, string, "[STRING:%.*s]", (int) tmp->len, tmp->val);
90  handlebars_talloc_free(tmp);
91  break;
92  case handlebars_operand_type_array: {
93  arr = operand->data.array.array;
94  string = handlebars_string_append(context, string, HBS_STRL("[ARRAY:"));
95  for( i = 0 ; i < operand->data.array.count; ++i ) {
96  if( i > 0 ) {
97  string = handlebars_string_append(context, string, HBS_STRL(","));
98  }

101  string = handlebars_string_append(context, string, HBS_STRL("]"));
102  break;
103  }

170 static void handlebars_opcode_printer_array_print(struct handlebars_opcode_printer * printer)
171 {
172  struct handlebars_opcode ** opcodes = printer->opcodes;
173  size_t count = printer->opcodes_length;
174  size_t i;
175 
176  for( i = 0; i < count; i++, opcodes++ ) {
177  printer->output = append_indent(CONTEXT, printer->output, printer->indent);
178  printer->output = handlebars_opcode_print_append(CONTEXT, printer->output, *opcodes, printer->flags);
179  printer->output = handlebars_string_append(CONTEXT, printer->output, HBS_STRL("\n"));
180  }
181 
182  printer->output = append_indent(CONTEXT, printer->output, printer->indent);
183  printer->output = handlebars_string_append(CONTEXT, printer->output, HBS_STRL("-----\n"));
184 }

186 static void handlebars_opcode_printer_print(struct handlebars_opcode_printer * printer, struct handlebars_program * program)
187 {
188  size_t i;
189  struct handlebars_program * child;
190 
191  // Print opcodes
192  printer->opcodes = program->opcodes;
193  printer->opcodes_length = program->opcodes_length;
194  handlebars_opcode_printer_array_print(printer);
195 
196  printer->indent++;
197 
198  // Print decorators
199  for( i = 0; i < program->decorators_length; i++ ) {
200  printer->output = append_indent(CONTEXT, printer->output, (size_t) printer->indent);
201  printer->output = handlebars_string_append(CONTEXT, printer->output, HBS_STRL("DECORATOR\n"));
202  child = *(program->decorators + i);
203  handlebars_opcode_printer_print(printer, child);
204  }
205 
206  // Print children
207  for( i = 0; i < program->children_length; i++ ) {
208  child = *(program->children + i);
209  handlebars_opcode_printer_print(printer, child);
210  }
211 
212  printer->indent--;
213 }
