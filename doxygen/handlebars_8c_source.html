217 struct handlebars_token ** handlebars_lex(struct handlebars_parser * parser)
218 {
219  YYSTYPE yylval_param;
220  YYLTYPE yylloc_param;
221  struct handlebars_error * e = HBSCTX(parser)->e;
222  jmp_buf * prev = e->jmp;
223  jmp_buf buf;
224  YYSTYPE * lval;
225  struct handlebars_token ** tokens;
226  struct handlebars_token * token;
227  size_t i = 0;
228 
229  // Save jump buffer
230  if( !prev ) {
231  e->jmp = &buf;
232  if (setjmp(buf)) {
233  goto done;
234  }
235  }
236 
237  // Prepare token list
238  tokens = MC(talloc_array(parser, struct handlebars_token *, 32));
239 
240  // Run
241  do {
242  int token_int = handlebars_yy_lex(&yylval_param, &yylloc_param, parser->scanner);
243  if( unlikely(token_int == END || token_int == INVALID) ) {
244  break;
245  }
246  lval = handlebars_yy_get_lval(parser->scanner);
247 
248  // Make token object
249  token = handlebars_token_ctor(HBSCTX(parser), token_int, lval->string);
250 
251  // Append
252  tokens = talloc_realloc(parser, tokens, struct handlebars_token *, i + 2);
253  tokens[i] = talloc_steal(tokens, token);
254  tokens[i + 1] = NULL;
255  i++;
256  } while( 1 );
257 
258 done:
259  e->jmp = prev;
260  return tokens;
261 }
