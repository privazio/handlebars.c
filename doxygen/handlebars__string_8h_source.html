51 static inline unsigned long handlebars_string_hash_cont(const char * str, size_t len, unsigned long hash)
52 {
53  unsigned long c;
54  const unsigned char * ptr = (const unsigned char *) str;
55  const unsigned char * end = ptr + len;
56  for( c = *ptr++; ptr <= end && c; c = *ptr++ ) {
57  hash = ((hash << 5) + hash) + c; /* hash * 33 + c */
58  }
59  return hash;
60 }
