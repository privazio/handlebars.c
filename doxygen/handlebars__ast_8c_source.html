180 const char * handlebars_ast_node_readable_type(int type)
181 {
182 #define _RTYPE_STR(str) #str
183 #define _RTYPE_MK(str) HANDLEBARS_AST_NODE_ ## str
184 #define _RTYPE_CASE(str, name) \
185  case _RTYPE_MK(str): return _RTYPE_STR(name); break
186  switch( type ) {
187  _RTYPE_CASE(NIL, NIL);
188  _RTYPE_CASE(BLOCK, block);
189  _RTYPE_CASE(BOOLEAN, BooleanLiteral);
190  _RTYPE_CASE(COMMENT, comment);
191  _RTYPE_CASE(CONTENT, content);
192  _RTYPE_CASE(HASH, hash);
193  _RTYPE_CASE(HASH_PAIR, HASH_PAIR);
194  _RTYPE_CASE(INTERMEDIATE, INTERMEDIATE);
195  _RTYPE_CASE(INVERSE, INVERSE);
196  _RTYPE_CASE(MUSTACHE, mustache);
197  _RTYPE_CASE(NUMBER, NumberLiteral);
198  _RTYPE_CASE(PARTIAL, partial);
199  _RTYPE_CASE(PARTIAL_BLOCK, PartialBlockStatement);
200  _RTYPE_CASE(PATH, PathExpression);
201  _RTYPE_CASE(PATH_SEGMENT, PATH_SEGMENT);
202  _RTYPE_CASE(PROGRAM, program);
203  _RTYPE_CASE(RAW_BLOCK, raw_block);
204  _RTYPE_CASE(SEXPR, SubExpression);
205  _RTYPE_CASE(STRING, StringLiteral);
206  _RTYPE_CASE(UNDEFINED, UNDEFINED);
207  case HANDLEBARS_AST_NODE_NUL: return "NULL";
