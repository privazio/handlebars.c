398 static void _handlebars_ast_print(struct handlebars_ast_node * ast_node, struct handlebars_ast_printer_context * ctx)
399 {
400  if( unlikely(ast_node == NULL) ) {

402  }

404  switch( ast_node->type ) {

408  return _handlebars_ast_print_block(ast_node, ctx);
409  case HANDLEBARS_AST_NODE_COMMENT:
410  return _handlebars_ast_print_comment(ast_node, ctx);
411  case HANDLEBARS_AST_NODE_CONTENT:
412  return _handlebars_ast_print_content(ast_node, ctx);
413  case HANDLEBARS_AST_NODE_HASH:
414  return _handlebars_ast_print_hash(ast_node, ctx);
415  case HANDLEBARS_AST_NODE_MUSTACHE:
416  return _handlebars_ast_print_mustache(ast_node, ctx);
417  case HANDLEBARS_AST_NODE_NUL:
418  return _handlebars_ast_print_null(ast_node, ctx);
419  case HANDLEBARS_AST_NODE_NUMBER:
420  return _handlebars_ast_print_number(ast_node, ctx);
421  case HANDLEBARS_AST_NODE_PARTIAL:
422  return _handlebars_ast_print_partial(ast_node, ctx);
423  case HANDLEBARS_AST_NODE_PARTIAL_BLOCK:
424  return _handlebars_ast_print_partial_block(ast_node, ctx);
425  case HANDLEBARS_AST_NODE_PATH:
426  return _handlebars_ast_print_path(ast_node, ctx);
427  case HANDLEBARS_AST_NODE_PROGRAM:
428  return _handlebars_ast_print_program(ast_node, ctx);
429  case HANDLEBARS_AST_NODE_RAW_BLOCK:
430  return _handlebars_ast_print_block(ast_node, ctx);
431  case HANDLEBARS_AST_NODE_SEXPR:
432  return _handlebars_ast_print_sexpr(ast_node, ctx);
433  case HANDLEBARS_AST_NODE_STRING:
434  return _handlebars_ast_print_string(ast_node, ctx);
435  case HANDLEBARS_AST_NODE_UNDEFINED:
436  return _handlebars_ast_print_undefined(ast_node, ctx);
437  // LCOV_EXCL_START
438  // Note: these should never be printed, intermediate nodes
439  case HANDLEBARS_AST_NODE_INTERMEDIATE:
440  case HANDLEBARS_AST_NODE_INVERSE:
441  // Note: these are currently implemented within their parent

443  case HANDLEBARS_AST_NODE_PATH_SEGMENT:
444  case HANDLEBARS_AST_NODE_NIL:

448  }
449 }
